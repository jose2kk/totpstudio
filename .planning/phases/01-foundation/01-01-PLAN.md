import * as React from "react"
import { ThemeProvider as NextThemesProvider } from "next-themes"

export function ThemeProvider({
  children,

  return <NextThemesProvider {...props}>{children}</NextThemesProvider>
}

This is the thin client wrapper required because `layout.tsx` is a Server Component but `NextThemesProvider` needs client context. Source: RESEARCH.md Pattern 2.

**File 2 — `src/components/theme-toggle.tsx`:**
```tsx
"use client"

import * as React from "react"
import { Moon, Sun } from "lucide-react"
import { useTheme } from "next-themes"
import { Button } from "@/components/ui/button"

export function ThemeToggle() {
  const { setTheme, theme } = useTheme()

  return (
    <Button
      variant="ghost"
      size="icon"
      onClick={() => setTheme(theme === "light" ? "dark" : "light")}
    >
      <Sun className="h-[1.5rem] w-[1.3rem] dark:hidden" />
      <Moon className="hidden h-5 w-5 dark:block" />
      <span className="sr-only">Toggle theme</span>
    </Button>
  )
}
```
Two-state toggle per D-05 and D-06. Sun visible in light mode, Moon visible in dark mode. `sr-only` label "Toggle theme" per UI-SPEC copywriting contract. Source: RESEARCH.md Pattern 3.

**File 4 — Update `src/app/layout.tsx`:**
Replace the entire contents with:
```tsx
import type { Metadata } from "next"
import { Inter } from "next/font/google"
import "./globals.css"
import { ThemeProvider } from "@/components/theme-provider"

const inter = Inter({ subsets: ["latin"] })

export const metadata: Metadata = {
  title: "TOTP Studio",
  description: "Generate TOTP codes and QR codes for authenticator apps",
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body className={inter.className}>
        <ThemeProvider
          attribute="class"
          defaultTheme="system"
          enableSystem
          disableTransitionOnChange
        >
          {children}
        </ThemeProvider>
      </body>
    </html>
  )
}
```
Critical elements:
- `suppressHydrationWarning` on `<html>` — prevents React hydration mismatch from next-themes class injection. Required per STATE.md decision and RESEARCH.md Pitfall 1.
- `ThemeProvider` wraps children with `attribute="class"` (for Tailwind dark: variant), `defaultTheme="system"` and `enableSystem` (per UI-02: system-aware default), `disableTransitionOnChange` (prevents flash per RESEARCH.md anti-patterns).
- Inter font loaded via `next/font/google` per UI-SPEC typography contract.
- Metadata title "TOTP Studio", description "Generate TOTP codes and QR codes for authenticator apps" per UI-SPEC copywriting contract.
- Do NOT add `"use client"` to this file — it is a Server Component. The client boundary is in `theme-provider.tsx`.
