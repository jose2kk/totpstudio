```typescript
import { generateURI } from 'otplib'
import type { HashAlgorithm } from 'otplib'

export interface OtpauthUriParams {
  secret: string
  issuer: string
  account: string
  algorithm: HashAlgorithm
  digits: number
  period: number
}

export function buildOtpauthUri(params: OtpauthUriParams): string | null {
  if (!params.secret) return null
  const stripped = params.secret.trim().replace(/=/g, '')
  if (!stripped) return null
  return generateURI({
    issuer: params.issuer,
    label: params.account || 'Account',
    secret: stripped,
    algorithm: params.algorithm,
    digits: params.digits,
    period: params.period,
  })
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input -> URI generation | Issuer and account free-text inputs are passed to generateURI which URL-encodes them |
| URI -> QR code | URI string is passed to react-qr-code for SVG rendering |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Injection | generateURI (issuer/account params) | accept | generateURI URL-encodes all params; QR code renders SVG from encoded data; no HTML/JS injection path. Issuer/account are free-text by design per D-06. |
| T-03-02 | Information Disclosure | otpauth:// URI in DOM | accept | URI contains the secret which is already visible in the secret input. No new information exposure. All client-side, no network transmission. |
| T-03-03 | Denial of Service | Extremely long issuer/account strings | accept | QR code may become dense but react-qr-code handles gracefully. No server resources at risk — client-side only. |
</threat_model>
